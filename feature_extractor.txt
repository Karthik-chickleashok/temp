from typing import List, Dict, Any


class FeatureExtractor:
    """
    Extracts structured diagnostic signals from parsed DLT rows.
    """

    def extract(self, rows: List[Dict[str, Any]]) -> Dict[str, Any]:
        key_events = []
        error_events = []
        warning_events = []
        keywords = []
        missing_expected_events = []
        sequence_observations = []
        evidence_refs = []

        event_flags = {
            "find_service_seen": False,
            "offer_service_seen": False,
            "subscribe_eventgroup_seen": False,
            "subscribe_ack_seen": False,
            "timeout_seen": False,
            "lte_registered": False,
            "data_sent": False,
            "data_received": False,
            "data_dropped": False,
        }

        for row in rows:
            payload_raw = row.get("payload", "")
            payload = self._normalize_text(payload_raw)
            level = (row.get("level") or "").lower()

            # Keyword extraction
            self._extract_keywords(payload, keywords)

            # Event detection
            self._extract_events(payload, key_events, event_flags)

            # Error detection
            if "error" in level or any(x in payload for x in ["timeout", "failed", "error", "rejected", "unavailable", "drop"]):
                self._add_unique(error_events, payload_raw)
                self._collect_evidence(row, evidence_refs)

            # Warning detection
            if "warn" in level or "warning" in payload:
                self._add_unique(warning_events, payload_raw)

        # Missing events
        missing_expected_events = self._detect_missing_events(event_flags)

        # Sequence observations
        sequence_observations = self._build_sequence_observations(event_flags)

        # Domain detection
        candidate_domain = self._detect_domain(keywords)

        return {
            "key_events": key_events,
            "error_events": error_events,
            "warning_events": warning_events,
            "keywords": keywords,
            "missing_expected_events": missing_expected_events,
            "sequence_observations": sequence_observations,
            "candidate_domain": candidate_domain,
            "event_flags": event_flags,
            "evidence_refs": evidence_refs,
        }

    # ------------------ Helpers ------------------

    def _normalize_text(self, text: str) -> str:
        if not text:
            return ""
        return text.replace("\x00", "").lower().strip()

    def _add_unique(self, lst: List[str], value: str):
        if value not in lst:
            lst.append(value)

    def _extract_keywords(self, payload: str, keywords: List[str]):
        kw_list = [
            "gateway", "forward", "routing", "dropped", "received", "send", "publish",
            "modem", "sim", "apn", "attach", "detach", "registration", "bearer",
            "lte", "network", "signal", "backend", "data session",
            "someip", "findservice", "offerservice", "subscribeeventgroup", "subscribeack",
            "uds", "nrc", "diagnostic", "security",
            "doip",
            "boot", "startup", "init", "ready"
        ]

        for kw in kw_list:
            if kw in payload:
                self._add_unique(keywords, kw)

    def _extract_events(self, payload: str, key_events: List[str], flags: Dict):
        if "findservice" in payload:
            flags["find_service_seen"] = True
            self._add_unique(key_events, "FindService")

        if "offerservice" in payload:
            flags["offer_service_seen"] = True
            self._add_unique(key_events, "OfferService")

        if "subscribeeventgroup" in payload:
            flags["subscribe_eventgroup_seen"] = True
            self._add_unique(key_events, "SubscribeEventgroup")

        if "subscribeack" in payload:
            flags["subscribe_ack_seen"] = True
            self._add_unique(key_events, "SubscribeAck")

        if "timeout" in payload:
            flags["timeout_seen"] = True
            self._add_unique(key_events, "Timeout")

        if any(x in payload for x in ["send", "publish"]):
            flags["data_sent"] = True

        if "received" in payload:
            flags["data_received"] = True

        if any(x in payload for x in ["drop", "dropped"]):
            flags["data_dropped"] = True

        if any(x in payload for x in ["network attached", "attach complete", "bearer active", "pdp active"]):
            flags["lte_registered"] = True

    def _detect_missing_events(self, flags: Dict) -> List[str]:
        missing = []

        if flags["find_service_seen"] and not flags["offer_service_seen"]:
            missing.append("OfferService")

        if flags["subscribe_eventgroup_seen"] and not flags["subscribe_ack_seen"]:
            missing.append("SubscribeAck")

        if flags["data_sent"] and not flags["lte_registered"]:
            missing.append("LTE readiness")

        return missing

    def _build_sequence_observations(self, flags: Dict) -> List[str]:
        obs = []

        if flags["find_service_seen"] and not flags["offer_service_seen"]:
            obs.append("Client requested service but provider did not respond")

        if flags["data_received"] and not flags["data_sent"]:
            obs.append("Data received but not forwarded")

        if flags["data_sent"] and not flags["lte_registered"]:
            obs.append("Data send attempted before LTE/network ready")

        if flags["data_dropped"]:
            obs.append("Data dropped in gateway flow")

        return obs

    def _collect_evidence(self, row: Dict, evidence: List[Dict]):
        evidence.append({
            "line_no": row.get("line_no"),
            "timestamp": row.get("timestamp"),
            "payload": row.get("payload"),
        })

    def _detect_domain(self, keywords: List[str]) -> str:
        scores = {
            "SOME/IP": 0,
            "LTE": 0,
            "TELEMATICS": 0,
            "GATEWAY": 0,
            "UDS": 0,
            "DoIP": 0,
            "BOOT": 0,
        }

        for kw in keywords:
            if kw in ["findservice", "offerservice", "subscribeeventgroup"]:
                scores["SOME/IP"] += 2
            elif kw in ["lte", "attach", "bearer"]:
                scores["LTE"] += 2
            elif kw in ["modem", "backend"]:
                scores["TELEMATICS"] += 2
            elif kw in ["gateway", "forward", "routing"]:
                scores["GATEWAY"] += 2
            elif kw in ["uds", "nrc"]:
                scores["UDS"] += 2
            elif kw in ["doip"]:
                scores["DoIP"] += 2
            elif kw in ["boot", "startup"]:
                scores["BOOT"] += 2

        return max(scores, key=scores.get)

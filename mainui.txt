from __future__ import annotations

import json
import os
import tkinter as tk
from tkinter import filedialog, messagebox

from parser.dlt_parser import DLTParser
from feature_extractor.feature_extractor import FeatureExtractor


class CCU2AnalyzerUI:
    def __init__(self, root: tk.Tk) -> None:
        self.root = root
        self.root.title("CCU2 Diagnosis Tool")
        self.root.geometry("1100x700")

        self.file_path: str | None = None
        self.parsed_data = None
        self.feature_result = None

        self._build_ui()

    def _build_ui(self) -> None:
        top = tk.Frame(self.root)
        top.pack(fill=tk.X, padx=10, pady=8)

        self.file_label = tk.Label(
            top,
            text="No file selected",
            anchor="w",
            relief=tk.SUNKEN,
            padx=8
        )
        self.file_label.pack(side=tk.LEFT, fill=tk.X, expand=True)

        tk.Button(top, text="Browse", width=12, command=self.browse_file).pack(side=tk.RIGHT, padx=5)

        controls = tk.Frame(self.root)
        controls.pack(fill=tk.X, padx=10, pady=5)

        self.btn_parse = tk.Button(controls, text="Parse", width=14, state=tk.DISABLED, command=self.parse_file)
        self.btn_parse.pack(side=tk.LEFT, padx=5)

        self.btn_extract = tk.Button(controls, text="Extract Features", width=16, state=tk.DISABLED, command=self.extract_features)
        self.btn_extract.pack(side=tk.LEFT, padx=5)

        self.btn_save = tk.Button(controls, text="Save Report", width=14, state=tk.DISABLED, command=self.save_report)
        self.btn_save.pack(side=tk.LEFT, padx=5)

        self.btn_clear = tk.Button(controls, text="Clear", width=12, command=self.clear_output)
        self.btn_clear.pack(side=tk.LEFT, padx=5)

        self.output_text = tk.Text(self.root, wrap=tk.WORD)
        self.output_text.pack(fill=tk.BOTH, expand=True, padx=10, pady=10)

    def browse_file(self) -> None:
        path = filedialog.askopenfilename(
            title="Select log or JSON file",
            filetypes=[
                ("Supported files", "*.txt *.dlt *.json"),
                ("Text log files", "*.txt *.dlt"),
                ("JSON files", "*.json"),
                ("All files", "*.*"),
            ],
        )
        if not path:
            return

        self.file_path = path
        self.file_label.config(text=path)

        self.parsed_data = None
        self.feature_result = None

        self.btn_parse.config(state=tk.NORMAL)
        self.btn_extract.config(state=tk.DISABLED)
        self.btn_save.config(state=tk.DISABLED)

        self._set_output_text(f"Selected file:\n{path}\n")

    def parse_file(self) -> None:
        if not self.file_path:
            messagebox.showwarning("No file", "Please select a file first.")
            return

        try:
            ext = os.path.splitext(self.file_path)[1].lower()

            if ext == ".json":
                with open(self.file_path, "r", encoding="utf-8") as f:
                    self.parsed_data = json.load(f)
            else:
                parser = DLTParser()
                self.parsed_data = parser.parse_file(self.file_path)

            row_count = len(self.parsed_data) if isinstance(self.parsed_data, list) else 0
            self._set_output_text(
                f"Parse successful.\n\n"
                f"File: {self.file_path}\n"
                f"Total parsed rows: {row_count}\n"
            )

            self.btn_extract.config(state=tk.NORMAL)
            self.btn_save.config(state=tk.DISABLED)

        except Exception as e:
            messagebox.showerror("Parse Error", str(e))

    def extract_features(self) -> None:
        if not self.parsed_data:
            messagebox.showwarning("No data", "Please parse a file first.")
            return

        try:
            extractor = FeatureExtractor()
            self.feature_result = extractor.extract(self.parsed_data)

            self._set_output_text(json.dumps(self.feature_result, indent=2))
            self.btn_save.config(state=tk.NORMAL)

        except Exception as e:
            messagebox.showerror("Feature Extraction Error", str(e))

    def save_report(self) -> None:
        if not self.feature_result:
            messagebox.showwarning("No report", "Nothing to save yet.")
            return

        path = filedialog.asksaveasfilename(
            title="Save report",
            defaultextension=".json",
            filetypes=[("JSON files", "*.json"), ("All files", "*.*")],
        )
        if not path:
            return

        try:
            with open(path, "w", encoding="utf-8") as f:
                json.dump(self.feature_result, f, indent=2)

            messagebox.showinfo("Saved", f"Report saved successfully:\n{path}")

        except Exception as e:
            messagebox.showerror("Save Error", str(e))

    def clear_output(self) -> None:
        self.file_path = None
        self.parsed_data = None
        self.feature_result = None
        self.file_label.config(text="No file selected")
        self.btn_parse.config(state=tk.DISABLED)
        self.btn_extract.config(state=tk.DISABLED)
        self.btn_save.config(state=tk.DISABLED)
        self._set_output_text("")

    def _set_output_text(self, text: str) -> None:
        self.output_text.delete("1.0", tk.END)
        self.output_text.insert(tk.END, text)


if __name__ == "__main__":
    root = tk.Tk()
    app = CCU2AnalyzerUI(root)
    root.mainloop()

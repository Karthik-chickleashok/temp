import tkinter as tk
from tkinter import filedialog, messagebox
import json

from feature_extractor.feature_extractor import FeatureExtractor


def load_file():
    path = filedialog.askopenfilename(filetypes=[("JSON files", "*.json")])
    if not path:
        return

    try:
        with open(path, "r") as f:
            data = json.load(f)

        extractor = FeatureExtractor()
        result = extractor.extract(data)

        output_text.delete(1.0, tk.END)
        output_text.insert(tk.END, json.dumps(result, indent=2))

    except Exception as e:
        messagebox.showerror("Error", str(e))


def save_report():
    content = output_text.get(1.0, tk.END)

    path = filedialog.asksaveasfilename(defaultextension=".json")
    if not path:
        return

    with open(path, "w") as f:
        f.write(content)

    messagebox.showinfo("Saved", "Report saved successfully")


root = tk.Tk()
root.title("CCU2 Log Analyzer")
root.geometry("900x600")

frame = tk.Frame(root)
frame.pack(pady=10)

btn_load = tk.Button(frame, text="Load Parsed Log (JSON)", command=load_file)
btn_load.pack(side=tk.LEFT, padx=5)

btn_save = tk.Button(frame, text="Save Report", command=save_report)
btn_save.pack(side=tk.LEFT, padx=5)

output_text = tk.Text(root, wrap=tk.WORD)
output_text.pack(fill=tk.BOTH, expand=True)

root.mainloop()

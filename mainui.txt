from __future__ import annotations

import json
from pathlib import Path
import tkinter as tk
from tkinter import filedialog, messagebox

from parser.dlt_parser import DLTController
from parser.normalizer import DLTNormalizer
from feature_extractor.feature_extractor import FeatureExtractor
from requirement_evaluator import RequirementEvaluator


class CCU2AnalyzerUI:
    def __init__(self, root: tk.Tk) -> None:
        self.root = root
        self.root.title("CCU2 Diagnosis Tool")
        self.root.geometry("1280x820")

        self.controller = DLTController()

        self.input_mode: str | None = None
        self.selected_files: list[str] = []
        self.selected_folder: str | None = None

        self.raw_messages: list[tuple[str, list]] = []
        self.normalized_rows: list[dict] = []
        self.feature_output: dict | None = None
        self.requirement_output: dict | None = None

        self.query_var = tk.StringVar()

        self._build_ui()

    def _build_ui(self) -> None:
        top = tk.Frame(self.root)
        top.pack(fill=tk.X, padx=10, pady=8)

        self.selection_label = tk.Label(
            top,
            text="No input selected",
            anchor="w",
            relief=tk.SUNKEN,
            padx=8,
        )
        self.selection_label.pack(side=tk.LEFT, fill=tk.X, expand=True)

        browse_frame = tk.Frame(top)
        browse_frame.pack(side=tk.RIGHT)

        tk.Button(
            browse_frame,
            text="Browse File",
            width=12,
            command=self.browse_single_file,
        ).pack(side=tk.LEFT, padx=4)

        tk.Button(
            browse_frame,
            text="Browse Files",
            width=12,
            command=self.browse_multiple_files,
        ).pack(side=tk.LEFT, padx=4)

        tk.Button(
            browse_frame,
            text="Browse Folder",
            width=12,
            command=self.browse_folder,
        ).pack(side=tk.LEFT, padx=4)

        query_frame = tk.Frame(self.root)
        query_frame.pack(fill=tk.X, padx=10, pady=5)

        tk.Label(query_frame, text="Scope / Query:").pack(side=tk.LEFT)

        self.query_entry = tk.Entry(
            query_frame,
            textvariable=self.query_var,
            width=80,
        )
        self.query_entry.pack(side=tk.LEFT, padx=6, fill=tk.X, expand=True)

        controls = tk.Frame(self.root)
        controls.pack(fill=tk.X, padx=10, pady=5)

        self.btn_parse = tk.Button(
            controls,
            text="Parse + Normalize",
            width=18,
            state=tk.DISABLED,
            command=self.parse_selection,
        )
        self.btn_parse.pack(side=tk.LEFT, padx=5)

        self.btn_save_normalized = tk.Button(
            controls,
            text="Save Normalized JSON",
            width=18,
            state=tk.DISABLED,
            command=self.save_normalized_json,
        )
        self.btn_save_normalized.pack(side=tk.LEFT, padx=5)

        self.btn_extract_features = tk.Button(
            controls,
            text="Extract Features",
            width=16,
            state=tk.DISABLED,
            command=self.extract_features,
        )
        self.btn_extract_features.pack(side=tk.LEFT, padx=5)

        self.btn_save_features = tk.Button(
            controls,
            text="Save Features JSON",
            width=16,
            state=tk.DISABLED,
            command=self.save_features_json,
        )
        self.btn_save_features.pack(side=tk.LEFT, padx=5)

        self.btn_evaluate_requirements = tk.Button(
            controls,
            text="Evaluate Requirements",
            width=18,
            state=tk.DISABLED,
            command=self.evaluate_requirements,
        )
        self.btn_evaluate_requirements.pack(side=tk.LEFT, padx=5)

        self.btn_save_requirements = tk.Button(
            controls,
            text="Save Requirement JSON",
            width=18,
            state=tk.DISABLED,
            command=self.save_requirement_json,
        )
        self.btn_save_requirements.pack(side=tk.LEFT, padx=5)

        self.btn_search = tk.Button(
            controls,
            text="Search / Diagnose",
            width=16,
            state=tk.DISABLED,
            command=self.run_search_analysis,
        )
        self.btn_search.pack(side=tk.LEFT, padx=5)

        self.btn_clear = tk.Button(
            controls,
            text="Clear",
            width=10,
            command=self.clear_output,
        )
        self.btn_clear.pack(side=tk.LEFT, padx=5)

        self.output_text = tk.Text(self.root, wrap=tk.WORD)
        self.output_text.pack(fill=tk.BOTH, expand=True, padx=10, pady=10)

    # ------------------------------------------------------------------
    # Browse actions
    # ------------------------------------------------------------------

    def browse_single_file(self) -> None:
        path = filedialog.askopenfilename(
            title="Select one DLT file",
            filetypes=[
                ("DLT files", "*.dlt *.gz"),
                ("All files", "*.*"),
            ],
        )
        if not path:
            return

        self.input_mode = "single"
        self.selected_files = [path]
        self.selected_folder = None

        self._reset_after_selection()
        self.selection_label.config(text=f"Single file: {path}")
        self._set_output_text(f"Selected single file:\n{path}\n")

    def browse_multiple_files(self) -> None:
        paths = filedialog.askopenfilenames(
            title="Select multiple DLT files",
            filetypes=[
                ("DLT files", "*.dlt *.gz"),
                ("All files", "*.*"),
            ],
        )
        if not paths:
            return

        self.input_mode = "multiple"
        self.selected_files = list(paths)
        self.selected_folder = None

        self._reset_after_selection()
        summary = "\n".join(self.selected_files[:20])
        extra = ""
        if len(self.selected_files) > 20:
            extra = f"\n... and {len(self.selected_files) - 20} more file(s)"

        self.selection_label.config(
            text=f"Multiple files selected: {len(self.selected_files)}"
        )
        self._set_output_text(
            f"Selected multiple files ({len(self.selected_files)}):\n{summary}{extra}\n"
        )

    def browse_folder(self) -> None:
        folder = filedialog.askdirectory(title="Select folder")
        if not folder:
            return

        self.input_mode = "folder"
        self.selected_files = []
        self.selected_folder = folder

        self._reset_after_selection()
        self.selection_label.config(text=f"Folder: {folder}")
        self._set_output_text(f"Selected folder:\n{folder}\n")

    # ------------------------------------------------------------------
    # Parse + normalize
    # ------------------------------------------------------------------

    def parse_selection(self) -> None:
        if not self.input_mode:
            messagebox.showwarning(
                "No input",
                "Please select a file, files, or folder first.",
            )
            return

        try:
            grouped_messages: list[tuple[str, list]] = []

            if self.input_mode == "single":
                file_path = Path(self.selected_files[0])
                msgs = self.controller.parse_file(file_path)
                grouped_messages.append((file_path.name, msgs))

            elif self.input_mode == "multiple":
                for file_str in self.selected_files:
                    file_path = Path(file_str)
                    msgs = self.controller.parse_file(file_path)
                    grouped_messages.append((file_path.name, msgs))

            elif self.input_mode == "folder":
                files = self.controller.collect_supported_files_from_folder(
                    self.selected_folder
                )
                if not files:
                    messagebox.showwarning(
                        "No files",
                        "No supported .dlt/.gz files found in selected folder.",
                    )
                    return

                for file_path in files:
                    msgs = self.controller.parse_file(file_path)
                    grouped_messages.append((file_path.name, msgs))

            else:
                raise ValueError(f"Unsupported input mode: {self.input_mode}")

            self.raw_messages = grouped_messages
            self.normalized_rows = DLTNormalizer.normalize_grouped_messages(
                grouped_messages
            )
            self.feature_output = None
            self.requirement_output = None

            total_files = len(grouped_messages)
            total_rows = len(self.normalized_rows)
            source_names = [name for name, _ in grouped_messages]

            source_preview = "\n".join(source_names[:20])
            if len(source_names) > 20:
                source_preview += f"\n... and {len(source_names) - 20} more file(s)"

            preview = self.normalized_rows[:20]

            output = (
                f"Parse + Normalize successful.\n\n"
                f"Input mode: {self.input_mode}\n"
                f"Total source files: {total_files}\n"
                f"Total normalized rows: {total_rows}\n\n"
                f"Source files:\n{source_preview}\n\n"
                f"Preview (first 20 normalized rows):\n\n"
                f"{json.dumps(preview, indent=2, ensure_ascii=False)}"
            )

            self._set_output_text(output)

            self.btn_save_normalized.config(state=tk.NORMAL)
            self.btn_extract_features.config(state=tk.NORMAL)
            self.btn_save_features.config(state=tk.DISABLED)
            self.btn_evaluate_requirements.config(state=tk.DISABLED)
            self.btn_save_requirements.config(state=tk.DISABLED)
            self.btn_search.config(state=tk.DISABLED)

        except Exception as e:
            messagebox.showerror("Parse Error", str(e))

    # ------------------------------------------------------------------
    # Feature extraction
    # ------------------------------------------------------------------

    def extract_features(self) -> None:
        if not self.normalized_rows:
            messagebox.showwarning(
                "No normalized data",
                "Please parse and normalize input first.",
            )
            return

        try:
            extractor = FeatureExtractor()
            self.feature_output = extractor.extract(self.normalized_rows)
            self.requirement_output = None

            self._set_output_text(
                json.dumps(self.feature_output, indent=2, ensure_ascii=False)
            )

            self.btn_save_features.config(state=tk.NORMAL)
            self.btn_evaluate_requirements.config(state=tk.NORMAL)
            self.btn_save_requirements.config(state=tk.DISABLED)
            self.btn_search.config(state=tk.DISABLED)

        except Exception as e:
            messagebox.showerror("Feature Extraction Error", str(e))

    # ------------------------------------------------------------------
    # Requirement evaluation
    # ------------------------------------------------------------------

    def evaluate_requirements(self) -> None:
        if not self.normalized_rows:
            messagebox.showwarning(
                "No normalized data",
                "Please parse and normalize input first.",
            )
            return

        if not self.feature_output:
            messagebox.showwarning(
                "No features",
                "Please extract features first.",
            )
            return

        try:
            evaluator = RequirementEvaluator()
            self.requirement_output = evaluator.evaluate(
                self.normalized_rows,
                self.feature_output,
            )

            display_data = {
                "requirement_features": self.requirement_output,
                "feature_count": len(self.requirement_output.get("results", []))
                if isinstance(self.requirement_output, dict)
                else 0,
            }

            self._set_output_text(
                json.dumps(display_data, indent=2, ensure_ascii=False)
            )
            self.btn_save_requirements.config(state=tk.NORMAL)
            self.btn_search.config(state=tk.NORMAL)

        except Exception as e:
            messagebox.showerror("Requirement Evaluation Error", str(e))

    # ------------------------------------------------------------------
    # Search / Diagnose
    # ------------------------------------------------------------------

    def run_search_analysis(self) -> None:
        query = self.query_var.get().strip().lower()

        if not query:
            messagebox.showwarning(
                "No query",
                "Please enter a scope or question.",
            )
            return

        if not self.requirement_output:
            messagebox.showwarning(
                "No requirement results",
                "Please evaluate requirements first.",
            )
            return

        try:
            result = self.search_reasoner(query)
            self._set_output_text(json.dumps(result, indent=2, ensure_ascii=False))

        except Exception as e:
            messagebox.showerror("Search / Diagnose Error", str(e))

    def search_reasoner(self, query: str) -> dict:
        categories = self._map_query_to_categories(query)

        results = self.requirement_output.get("results", []) if isinstance(
            self.requirement_output, dict
        ) else []

        if categories:
            filtered = [
                r for r in results
                if str(r.get("category", "")).upper() in categories
            ]
        else:
            filtered = results

        failed = [r for r in filtered if r.get("status") == "FAIL"]
        passed = [r for r in filtered if r.get("status") == "PASS"]
        not_applicable = [r for r in filtered if r.get("status") == "NOT_APPLICABLE"]

        conclusion = self._build_search_conclusion(
            query=query,
            categories=categories,
            failed=failed,
            passed=passed,
            not_applicable=not_applicable,
        )

        return {
            "query": query,
            "selected_categories": categories if categories else ["ALL"],
            "summary": {
                "total_matching_requirements": len(filtered),
                "failed": len(failed),
                "passed": len(passed),
                "not_applicable": len(not_applicable),
            },
            "failed_results": failed[:20],
            "passed_results": passed[:20],
            "conclusion": conclusion,
        }

    def _map_query_to_categories(self, query: str) -> list[str]:
        q = query.lower()
        categories: list[str] = []

        mapping = {
            "APN": ["apn"],
            "NETWORK": ["network", "internet", "ip", "connectivity"],
            "GATEWAY": ["gateway", "forward", "routing", "route"],
            "CONTROL": ["someip", "some/ip", "request", "response", "control"],
            "SIM": ["sim", "imsi", "iccid", "eid"],
            "GENERAL": ["timeout", "error", "warn", "general"],
        }

        for category, tokens in mapping.items():
            if any(token in q for token in tokens):
                categories.append(category)

        deduped: list[str] = []
        for category in categories:
            if category not in deduped:
                deduped.append(category)

        return deduped

    def _build_search_conclusion(
        self,
        query: str,
        categories: list[str],
        failed: list[dict],
        passed: list[dict],
        not_applicable: list[dict],
    ) -> str:
        if not categories:
            if failed:
                return (
                    f"Query '{query}' did not map to a specific category, but "
                    f"{len(failed)} failed requirement(s) were found in the full result set."
                )
            return (
                f"Query '{query}' did not map to a specific category and no clear failures "
                f"were found in the current evaluated results."
            )

        if failed:
            failed_names = ", ".join(
                str(item.get("feature_name", "")) for item in failed[:5]
            )
            return (
                f"For query '{query}', category filter {categories} found "
                f"{len(failed)} failed requirement(s). "
                f"Top failed checks: {failed_names}."
            )

        if passed:
            return (
                f"For query '{query}', category filter {categories} found no failures. "
                f"{len(passed)} requirement(s) passed."
            )

        return (
            f"For query '{query}', category filter {categories} found only "
            f"NOT_APPLICABLE results or no matching requirement evaluations."
        )

    # ------------------------------------------------------------------
    # Save actions
    # ------------------------------------------------------------------

    def save_normalized_json(self) -> None:
        if not self.normalized_rows:
            messagebox.showwarning(
                "No parsed data",
                "Please parse input first.",
            )
            return

        path = filedialog.asksaveasfilename(
            title="Save normalized JSON",
            defaultextension=".json",
            filetypes=[("JSON files", "*.json"), ("All files", "*.*")],
        )
        if not path:
            return

        try:
            with open(path, "w", encoding="utf-8") as f:
                json.dump(self.normalized_rows, f, indent=2, ensure_ascii=False)

            messagebox.showinfo(
                "Saved",
                f"Normalized JSON saved successfully:\n{path}",
            )

        except Exception as e:
            messagebox.showerror("Save Error", str(e))

    def save_features_json(self) -> None:
        if not self.feature_output:
            messagebox.showwarning(
                "No features",
                "Please extract features first.",
            )
            return

        path = filedialog.asksaveasfilename(
            title="Save features JSON",
            defaultextension=".json",
            filetypes=[("JSON files", "*.json"), ("All files", "*.*")],
        )
        if not path:
            return

        try:
            with open(path, "w", encoding="utf-8") as f:
                json.dump(self.feature_output, f, indent=2, ensure_ascii=False)

            messagebox.showinfo(
                "Saved",
                f"Features JSON saved successfully:\n{path}",
            )

        except Exception as e:
            messagebox.showerror("Save Error", str(e))

    def save_requirement_json(self) -> None:
        if self.requirement_output is None:
            messagebox.showwarning(
                "No requirement results",
                "Please evaluate requirements first.",
            )
            return

        path = filedialog.asksaveasfilename(
            title="Save requirement evaluation JSON",
            defaultextension=".json",
            filetypes=[("JSON files", "*.json"), ("All files", "*.*")],
        )
        if not path:
            return

        try:
            with open(path, "w", encoding="utf-8") as f:
                json.dump(self.requirement_output, f, indent=2, ensure_ascii=False)

            messagebox.showinfo(
                "Saved",
                f"Requirement JSON saved successfully:\n{path}",
            )

        except Exception as e:
            messagebox.showerror("Save Error", str(e))

    # ------------------------------------------------------------------
    # Helpers
    # ---------------------------------------------------------
# ------------------------------------------------------------------
    # Helpers
    # ------------------------------------------------------------------

    def clear_output(self) -> None:
        self.input_mode = None
        self.selected_files = []
        self.selected_folder = None
        self.raw_messages = []
        self.normalized_rows = []
        self.feature_output = None
        self.requirement_output = None
        self.query_var.set("")

        self.selection_label.config(text="No input selected")

        self.btn_parse.config(state=tk.DISABLED)
        self.btn_save_normalized.config(state=tk.DISABLED)
        self.btn_extract_features.config(state=tk.DISABLED)
        self.btn_save_features.config(state=tk.DISABLED)
        self.btn_evaluate_requirements.config(state=tk.DISABLED)
        self.btn_save_requirements.config(state=tk.DISABLED)
        self.btn_search.config(state=tk.DISABLED)

        self._set_output_text("")

    def _reset_after_selection(self) -> None:
        self.raw_messages = []
        self.normalized_rows = []
        self.feature_output = None
        self.requirement_output = None

        self.btn_parse.config(state=tk.NORMAL)
        self.btn_save_normalized.config(state=tk.DISABLED)
        self.btn_extract_features.config(state=tk.DISABLED)
        self.btn_save_features.config(state=tk.DISABLED)
        self.btn_evaluate_requirements.config(state=tk.DISABLED)
        self.btn_save_requirements.config(state=tk.DISABLED)
        self.btn_search.config(state=tk.DISABLED)

    def _set_output_text(self, text: str) -> None:
        self.output_text.delete("1.0", tk.END)
        self.output_text.insert(tk.END, text)


if __name__ == "__main__":
    root = tk.Tk()
    app = CCU2AnalyzerUI(root)
    root.mainloop()
